  inline Matrix *
  real (const CAbstMatrix& x)
  {
    Matrix *y = new Matrix(x.rows(), x.cols());
    for (Index k = 0; k < x.cols(); k++)
      for (Index j= 0; j < x.rows(); j++)
	(*y)(j,k) = real(x(j,k));
    return y;
  }

    for (Index k = 0; k < x.cols(); k++)

	(*y)(j,k) = imag(x(j,k));
    return y;
  }

  inline Matrix *
  conj (const CAbstMatrix& x)
  {
    Matrix *y = new Matrix(x);
    for (Index k = 0; k < x.size(); k++) (*y)(k) = conj((*y)(k));
    return y;
  }

  inline Real
  Matrix::frobnorm (void) const
  {
    char norm = 'F';
    int m = this->rows(), n = this->cols();
    Real t = F77_FUN(zlange)(&norm, &m, &n, this->data(), &m, 0);
    return t;
  }

  inline Real

    return t;
  }

  inline Real
  Matrix::norminf (void) const
  {
    try {
      char norm = 'I';
      int m = this->rows(), n = this->cols();
      Real *work = 0, t;
      try {
	work = new Real[m];
      } catch (std::bad_alloc) {
	throw OutOfMemory();
      }
      t = F77_FUN(zlange)(&norm, &m, &n, this->data(), &m, work);
      delete[] work;
      return t;
    } catch (Exception e) {
      std::cerr << e;
      throw Exception("matrix infinity norm");
    }
  }

  }

  /**
     Ordinary inner product with another vector.
  */
  inline Complex
  Matrix::dot (const Matrix& x) const
  {
    Complex d = 0;
    if (this->size() != x.size())
      throw DimError("complex dot product");
    CBLAS_FUN(zdotc_sub)(this->size(), this->data(), 1, x.data(), 1, &d);
    return d;
  }

  /**
     Find the roots of a single-variable polynomial of degree n,
     the coefficients of which are stored in the matrix x, which
     is in column-vector format and has length n+1. 
     The polynomial corresponding to a is 
     a(0) x^n + a(1) x^(n-1) + ... + a(n-2) x + a(n).
  */
  inline Matrix *
  roots (const Matrix& a)
  {
    Matrix *c = companion(a);
    Matrix *w = eigenvals(*c);
    delete c;
    return w;
  }

  inline Matrix&
  Matrix::operator = (Random& r)
  {
    for (Index k = 0; k < this->size(); k++)
      (*this)(k) = Complex(r(), r());
    return *this;
  }

  inline Submatrix&
  Submatrix::operator = (Random& r)
  {
    for (Index j = 0; j < this->rows(); j++) 
      for (Index k = 0; k < this->cols(); k++) 
	(*this)(j,k) = Complex(r(), r());

    return *this;
