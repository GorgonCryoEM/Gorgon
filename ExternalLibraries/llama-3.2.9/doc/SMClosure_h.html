// -*- C++ -*-

#ifndef _LLAMA_SMCLOSURE_H_
#define _LLAMA_SMCLOSURE_H_

#include "Decls.h"
#include "General.h"
#include "Exception.h"
#include "Matrix.h"

namespace Llama {

  /**
     Scalar-matrix multiplication closure.
  */

  template 
  class SMClosure {

  private:

    Scalar _a;
    const Matrix *_v;

    Scalar alpha (void) const {
      return _a;
    };

    const Matrix *matrix (void) const {
      return _v;
    };

    Index size (void) const {
      return this->matrix()->size();
    };

    Index rows (void) const {
      return this->matrix()->rows();
    };

    Index cols (void) const {
      return this->matrix()->cols();
    };

    SMClosure (void) : _a(0), _v(0) { };

    SMClosure& operator = (const SMClosure&);

  public:

    SMClosure (const Scalar a, const Matrix *v) : _a(a), _v(v) { };

    SMClosure (const SMClosure& x) : _a(x.alpha()), _v(x.matrix()) { };

    ~SMClosure (void) { };

    friend class Matrix;

  };

  // Rules of scalar/matrix arithmetic

  template 
  inline SMClosure
  operator * (const Matrix& x, const Scalar a)
  {
    return SMClosure(a, &x);
  }

  template 
  inline SMClosure
  operator * (const Scalar a, const Matrix& x)
  {
    return SMClosure(a, &x);
  }

  template 
  inline SMClosure
  operator * (const Scalar a, const SMClosure& x)
  {
    return SMClosure(a * x.alpha(), x.matrix());
  }

  template 
  inline SMClosure
  operator * (const SMClosure& x, const Scalar a)
  {
    return SMClosure(x.alpha() * a, x.matrix());
  }

  inline SMClosure
  operator * (const Real a, const Matrix& x)
  {
    return SMClosure(a, &x);
  }

  inline SMClosure
  operator * (const Matrix& x, const Real a)
  {
    return SMClosure(a, &x);
  }

}

#endif
